g++ -O2 -std=c++17 "$3".cpp -o $3
for ((i = 1;; i++)); do
  echo 'Test #'$i
  timeout 5s ./$3 $RANDOM > in
  timeout 5s ./$1 < in > out
  timeout 5s ./$2 < in > ans
  diff -yi ans out > diff.out

    cat diff.out
    break
  fi
done
